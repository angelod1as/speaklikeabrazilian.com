# create the following directories

sudo mkdir -p /var/log/searchd/
sudo mkdir -p /var/data/sphinx/
sudo chown sphinxsearch:sphinxsearch /var/log/searchd/ -R
sudo chown sphinxsearch:sphinxsearch /var/data/sphinx/ -R

# copy sphinxd to /etc/init.d

sudo update-rc.d sphinxd defaults
sudo /etc/init.d/sphinxd <start|stop|restart>
